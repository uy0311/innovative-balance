	ib_scorched_earth = {
		icon = generic_scorched_earth

		state_target = yes

		on_map_mode = map_only

		visible = {
			ROOT = {
				has_war = yes
		#		NOT = { has_country_flag = ib_se_auto_on_flag }
			}
			FROM = {
				state_population > 100
				NOT = { has_state_flag = ib_scorched_earth_state_flag }
				NOT = {
					all_neighbor_state = {
						controller = {
							OR = {
								tag = ROOT
								is_subject_of = ROOT
								is_in_faction_with = ROOT
							}
						}
					}
				}
				any_neighbor_state = {
					if = {
						limit = {
							any_country = {
								controls_state = PREV
								has_war = yes
								tag = ROOT
							}
						}
						controller = {
							has_war = yes
							NOT = {
								tag = ROOT
							}
						}
					}
				}
				if = {
					limit = {
						any_country = {
							controls_state = PREV
							tag = ROOT
						}
					}
					controller = {
						tag = ROOT
					}
				}
				else = {
					always = no
				}
				NOT = { has_dynamic_modifier = { modifier = ib_scorched_earth_modifier } }
			}
		}

		available = {
			custom_trigger_tooltip = {
				tooltip = ib_se_trigger_tt
				FROM = {
					NOT = { has_dynamic_modifier = { modifier = ib_scorched_earth_ready } }
					NOT = { has_dynamic_modifier = { modifier = ib_scorched_earth_modifier } }
				}
				OR = {
					AND = {
						divisions_in_state = {
							type = infantry
							state = FROM 
							size > 4 
						}
						army_manpower_in_state = { 
							state = FROM 
							amount > 50000 
						}
						check_variable = { num_equipment_in_armies_k@infantry_equipment > 2.5 }
						check_variable = { num_equipment_in_armies@support_equipment > 35 }
					}
					any_allied_country = {
						AND = {
							divisions_in_state = {
								type = infantry
								state = FROM 
								size > 4 
							}
							army_manpower_in_state = { 
								state = FROM 
								amount > 50000 
							}
							check_variable = { num_equipment_in_armies_k@infantry_equipment > 2.5 }
							check_variable = { num_equipment_in_armies@support_equipment > 35 }
						}
					}
				}
			}
			
		}

		custom_cost_trigger = {

		}

		custom_cost_text = decision_cost_st_3

	#	cost = 50
		fire_only_once = no

        days_remove = 1
		days_re_enable = 30

				}
				add_stability = -0.03
				add_offsite_building = { type = industrial_complex level = 1 }
			}
		}

		remove_effect = {
			hidden_effect = {
				add_offsite_building = { type = industrial_complex level = -1 }
			}
		}
